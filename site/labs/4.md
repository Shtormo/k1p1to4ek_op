    std::cout << "-----------------------------------------------------------" << std::endl; // для прикраси


    int numnum; // змінна , яка буде використовуватися для збереження кількості дробів у ланцюговому дробі.
    std::cout << "Введіть кількість дробів у ланцюговому дробі для кореня квадратного 2: ";
    std::cin >> numnum; // Зчитує введене користувачем число дробів у ланцюговому дробі та зберігає його у змінній

    double result = 0.0; // Змінні використовуються для обчислення кореня квадратного з числа 2 за допомогою ланцюгового дробу. result - проміжний результат, koren_2 - кінцевий результат.
    for (int g = 0; g < numnum; g++) { // Цей цикл використовується для обчислення ланцюгового дробу з заданою кількістю додатних дробів.
        if (g == 0) { // У циклі обчислення результату ланцюгового дробу відбувається за формулою, яка залежить від парності "g".
            result = 1.0;
        }
        else { // Якщо g не дорівнює 0, тобто це не перший член ланцюгового дробу, виконується ця частина коду.
            if (g % 2 == 0) { // В цьому розділі перевіряється, чи g - парне число (залишок від ділення на 2 дорівнює 0). Якщо g парне, то виконується цей блок коду.
                result = 2.0 + 1.0 / result; //  Якщо g - парне число, то в result обчислюється наступний член ланцюгового дробу. Ця формула відповідає правилам обчислення ланцюгового дробу для кореня квадратного з числа 2.
            }
            else { // g - непарне число, виконується ця частина коду.
                result = 1.0 + 1.0 / result; // Якщо g - непарне число, то в result обчислюється інший член ланцюгового дробу. Ця формула також відповідає правилам обчислення ланцюгового дробу для кореня квадратного з числа 2.
            }
        }
    }

    double koren_2 = 1.0 + 1.0 / result;

    std::cout << "Результат ланцюгового дробу для кореня квадратного 2: " << koren_2 << std::endl;

    return 0;
}
